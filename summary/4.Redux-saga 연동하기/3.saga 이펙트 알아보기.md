```javascript
import { all, fork, call, take, put } from 'redux-saga/effects';
import axios from 'axios';

function loginAPI() {
  return axios.post('/api/login');
}

function* logIn() {
  const result = yield call(loginAPI);
    try {
      yield put({
        type: 'LOG_IN_SUCCESS',
        data: result.data
      });
    } catch (err) {
      yield put({
        type: 'LOG_IN_FAILURE',
        // 실해한 데이터의 자료는 err.response.data에 담긴다.
        data: err.response.data 
      });
    }
}

function* watchLogin() {
  // take LOG_IN이라는 action이 실행될때까지 기다린다.
  yield take('LOG_IN');
}

function* watchLogOut() {
  yield take('LOG_OUT');
}

function* watchAddPost() {
  yield take('ADD_POST');
}

export default function* rootSaga() {
  yield all([
  	fork(watchLogin), 
    fork(watchLogOut), 
    fork(watchAddPost)
  ]);
}
```
put : fetch의 역할

call과 fork의 차이점
- fork는 비동기 처리를 해준다.
```javascript
function* logIn() {
	try {
		const result = yield fork(logInAPI);
		
		// fork 처리시 axios의 호출을 기다리지 않고 그대로 넘어간다.
		axios.post('/api/login')
		yield put({
			type: 'LOG_IN_SUCCESS',
			data: result.data
		})
  } catch (err) {
		
  }
}
```

- call은 동기 처리를 해준다.
```javascript
function* logIn() {
	try {
		const result = yield call(logInAPI);
		
		// call 처리시 axios then을 처리해주는것과 동일하다.
		axios.post('/api/login')
      .then(
				yield put({
					type: 'LOG_IN_SUCCESS',
					data: result.data
				})  	
      )
  } catch (err) {
		
  }
}
```
